```
/*Question:
输入一个链表，输出该链表中倒数第k个结点。
*/

/*Method:
 为了实现只遍历链表一次就能找到第k个结点，可以定义两个指针。
 第一个指针从链表的头指针开始遍历向前走k-1步，第二个指针保持不动。
 从第k步开始，第二个指针也开始从链表的头指针开始遍历。由于两个指针的距离保持在k-1，当第一个指针到达链表的尾结点时，第二个指针正好是倒数第k个结点。
*/
```
```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func getKthFromEnd(head *ListNode, k int) *ListNode {
	if k <= 0{
		return nil
	}
	
	fast := head
	for i := 1; i < k && fast != nil; i++{
		fast = fast.Next
	}
	
	if fast == nil{
		return nil
	}
	
	result := head
	for fast.Next != nil{
		result, fast = result.Next, fast.Next
	}
	
	return result
}
```

```cpp
//Code:
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/

class Solution {
public:
    ListNode* FindKthToTail(ListNode* pListHead, unsigned int k) {
    	if(pListHead == nullptr || k <= 0) return nullptr;
        
        ListNode* pFast = pListHead;
        for(int i = 0; i < k-1; ++i){
        	if(pFast->next != nullptr){
            	pFast = pFast->next;
            }
            //k大于链表长度
            else return nullptr;
        }
        
        ListNode* pSlow = pListHead;    
        while(pFast->next != nullptr){
            pSlow = pSlow->next;
            pFast = pFast->next;
        }
        
        return pSlow;
    }
};
```
